Z=(len(L))/2
ZI=floor(Z)

firsthalf=[]
secondhalf=[]

for i in range(ZI):
    firsthalf.append(L[i])

    
for j in range(ZI):
    secondhalf.append(L[len(L)-j-1])


print('Q1 is: '+`median(firsthalf)`)
print('Q3 is: '+`median(secondhalf)`)

\begin{enumerate}
\item Mean: $\answer{2.9167}$

\begin{multipleChoice}

\end{enumerate}


\end{problem}
