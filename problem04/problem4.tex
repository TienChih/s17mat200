\begin{abstract}
      
In this activity, we are given an example problem and we try to identify what we know, what we're trying to do, and how to approach it.
      
\end{abstract}
      
\maketitle
      
 
Out of a group of people who were subjected to a drug trial, 60\% lost teeth, 40\% lost hair, 25\% lost both teeth and hair.  How many lost either teeth or hair?  Let $T$ denote ``lost teeth" and $H$ denote ``lost hair."

 
 \begin{enumerate}
\item What do we know?

\begin{selectAll}
\choice[correct]{$P(T)$}
\choice[correct]{$P(H)$}
\choice[correct]{$P(T\ \text{and}\ H)$}
\choice{$P(T\ \text{or}\ H)$}
\choice{$P(H|T$)}
\choice{$P(H|T^c$)}
\choice{$P(T|H)$}
\choice{$P(T|H^c)$}
\end{selectAll}

\item What are we trying to find?

\begin{multipleChoice}
\choice{$P(T)$}
\choice{$P(H)$}
\choice{$P(T^c)$}
\choice{$P(H^c)$}
\choice{$P(T\ \text{and}\ H)$}
\choice{$P(T^c\ \text{and}\ H)$}
\choice{$P(T\ \text{and}\ H^c)$}
\choice{$P(T^c\ \text{and}\ H^c)$}

\choice[correct]{$P(T\ \text{or}\ H)$}
\choice{$P(T^c\ \text{or}\ H)$}
\choice{$P(T\ \text{or}\ H^c)$}
\choice{$P(T^c\ \text{or}\ H^c)$}

\choice{$P(H|T)$}
\choice{$P(H^c|T)$}
\choice{$P(H|T^c)$}
\choice{$P(H^c|T^c)$}

\choice{$P(T|H)$}
\choice{$P(T^c|H)$}
\choice{$P(T|H^c)$}
\choice{$P(T^c|H^c)$}

\end{multipleChoice}

\item The most relevant formula is:

\begin{multipleChoice}
\choice[correct]{$P(A\ \text{or}\ B)=P(A)+P(B)-P(A\ \text{and}\ B)$}
\choice{$P(A|B)=\frac{P(A\ \text{and}\ B)}{P(B)}$}
\choice{$P(B)=P(B|A)P(A)+P(B|A^c)(1-P(A))$}
\end{multipleChoice}

\item What is the actual answer?  The probability is (round to 4 decimal places if neccesary): $\answer{0.8}$

\item[Bonus:] If someone didn't lose their teeth, what is the probability they didn't lose their hair?  The probability is (round to 4 decimal places if neccesary): $\answer{0.625}$.

\end{enumerate}
