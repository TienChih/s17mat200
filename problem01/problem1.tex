\begin{abstract}
      
In this activity, we are given an example problem and we try to identify what we know, what we're trying to do, and how to approach it.
      
\end{abstract}
      
\maketitle
      
 
Suppose that in a math program, everyone who writes an expository paper or presents at a conference is given a participation award by the department.  Those who do both are given a special honors award.  In 2017, 60\% of the graduates earn the participation award, and 30\% of them earn the honors award.  If 40\% of graduates wrote expository papers, how many presented at a conference? Let $E$ be the event ``wrote an expository paper" and $C$ denote ``presented at a conference".
 
 \begin{enumerate}
\item What do we know?

\begin{selectAll}
\choice[correct]{$P(E)$}
\choice{$P(C)$}
\choice[correct]{$P(E\ \text{and}\ C)$}
\choice[correct]{$P(E\ \text{or}\ C)$}
\choice{$P(C|E$)}
\choice{$P(C|E^c$)}
\choice{$P(E|C)$}
\choice{$P(E|C^c)$}
\end{selectAll}

\item What are we trying to find?

\begin{multipleChoice}
\choice{$P(E)$}
\choice[correct]{$P(C)$}
\choice{$P(E^c)$}
\choice{$P(C^c)$}
\choice{$P(E\ \text{and}\ C)$}
\choice{$P(E^c\ \text{and}\ C)$}
\choice{$P(E\ \text{and}\ C^c)$}
\choice{$P(E^c\ \text{and}\ C^c)$}

\choice{$P(E\ \text{or}\ C)$}
\choice{$P(E^c\ \text{or}\ C)$}
\choice{$P(E\ \text{or}\ C^c)$}
\choice{$P(E^c\ \text{or}\ C^c)$}

\choice{$P(C|E)$}
\choice{$P(C^c|E)$}
\choice{$P(C|E^c)$}
\choice{$P(C^c|E^c)$}

\choice{$P(E|C)$}
\choice{$P(E^c|C)$}
\choice{$P(E|C^c)$}
\choice{$P(E^c|C^c)$}

\end{multipleChoice}

\item The most relevant formula is:

\begin{multipleChoice}
\choice[correct]{$P(A\ \text{or}\ B)=P(A)+P(B)-P(A\ \text{and}\ B)$}
\choice{$P(A|B)=\frac{P(A\ \text{and}\ B)}{P(B)}$}
\choice{$P(B)=P(B|A)P(A)+P(B|A^c)(1-P(A))$}
\end{multipleChoice}

\item What is the actual answer?  The probability is (round to 4 decimal places if neccesary): $\answer{0.5}$

\item[Bonus:] If someone didn't write an expository paper, what was the probability that they presented at a conference?  The probability is (round to 4 decimal places if neccesary): $\answer{0.75}$.

\end{enumerate}
